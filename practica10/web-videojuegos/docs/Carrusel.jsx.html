import { useState } from 'react';
import { useLoaderData } from 'react-router-dom';

/**
 * Componente de Carrusel interactivo para mostrar juegos populares.
 * Extrae los datos autom√°ticamente del loader de la ruta actual.
 * * @component
 * @example
 * <Carrusel />
 */
function Carrusel() {
    /** * @type {Object[]} 
     * Lista de objetos con los datos de los videojuegos.
     */
    const populares = useLoaderData(); 

    /** * @type {Array} 
     * Estado del √≠ndice actual y su setter [valor, funci√≥n].
     */
    const [indice, setIndice] = useState(0);

    /**
     * Avanza al siguiente juego en el array.
     */
    const siguiente = () => setIndice((indice + 1) % populares.length);

    /**
     * Retrocede al juego anterior en el array.
     */
    const anterior = () => setIndice((indice - 1 + populares.length) % populares.length);

    if (!populares || populares.length === 0) return null;

    /** * @type {Object} 
     * Datos del juego que se muestra actualmente. 
     */
    const juego = populares[indice];

    return (
        <div className="relative w-full max-w-5xl mx-auto h-96 bg-gray-900 rounded-3xl overflow-hidden shadow-2xl">
            
            {/* Imagen de fondo del juego actual */}
            <img 
                src={juego.background_image} 
                alt={juego.name} 
                className="absolute inset-0 w-full h-full object-cover opacity-50 transition-opacity duration-500"
            />
            
            {/* Informaci√≥n del juego (Texto y Degradado) */}
            <div className="relative flex flex-col justify-end h-full p-10 text-white bg-gradient-to-t from-black/90 to-transparent">
                <div className="z-10">
                    <span className="bg-yellow-500 text-black px-4 py-1 rounded-full text-sm font-bold">
                        Tendencia üî•
                    </span>
                    <h2 className="text-5xl font-black mt-4 drop-shadow-lg">{juego.name}</h2>
                    <div className="flex gap-4 mt-2 text-xl opacity-90">
                        <p>‚≠ê {juego.rating}</p>
                        <p>üìÖ {juego.released}</p>
                    </div>
                </div>
            </div>

            {/* Botones de navegaci√≥n */}
            <button 
                onClick={anterior} 
                className="z-20 absolute left-5 top-1/2 -translate-y-1/2 bg-white/10 hover:bg-white/30 text-white p-4 rounded-full backdrop-blur-sm transition-all"
            >
                ‚ùÆ
            </button>
            <button 
                onClick={siguiente} 
                className="z-20 absolute right-5 top-1/2 -translate-y-1/2 bg-white/10 hover:bg-white/30 text-white p-4 rounded-full backdrop-blur-sm transition-all"
            >
                ‚ùØ
            </button>
        </div>
    );
}

export default Carrusel;
